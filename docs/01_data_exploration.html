lter_datetime <- function(ym, dhm) {
  year2 <- as.numeric(substr(ym, 1, 2))
  year4 <- ifelse(year2 > 90, 1900 + year2, 2000 + year2)
  dhm6 <- ifelse(str_length(dhm) == 5, paste0("0", dhm), dhm)
  ymd_hm(paste0(year4, substr(ym, 3, 4), dhm6), tz = "UTC")
}
